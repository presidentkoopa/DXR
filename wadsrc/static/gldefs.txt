// Only 3 valid postprocess targets exist in this engine (gldefs.cpp:1526-1528): beforebloom,
// scene, screen. Both of these are full-screen overlay effects that sample the already-rendered
// image (InputTexture/TexCoord/ScreenSize) -- "screen" is the correct target, not their original
// invalid custom names. Also added the explicit "name" field (gldefs.cpp:1546-1549) -- PPShader.
// SetEnabled/SetUniform1f (vr_rom_load.zs, vr_death_effects.zs) match against shader.Name, a
// separate field from Target that was never set at all in the original entries, so those ZScript
// enable/disable calls were already non-functional before this fix, for a different reason.
HardwareShader PostProcess screen
{
	Name "rom_load"
	Shader "shaders/pp/rom_load.fp" 330
	// scanline_pos is set from ZScript (vr_rom_load.zs SetUniform1f). It MUST be declared here so
	// the engine registers it in the shader's Uniforms map and auto-emits `uniform float
	// scanline_pos;` into the compiled prolog -- without this line the .fp references an undeclared
	// identifier and the GLSL compile HARD-CRASHES the game on new-game load.
	uniform float scanline_pos
}

HardwareShader PostProcess screen
{
	Name "fatal_exception"
	Shader "shaders/pp/fatal_exception.fp" 330
	// Same reason: u_FatalStrength is set from ZScript (vr_death_effects.zs) -- declare it here or
	// the shader crashes the same way when the player dies.
	uniform float u_FatalStrength
}
