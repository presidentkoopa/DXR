// ============================================================================
//  DoomPlayer body avatar -- the first-person VR "OUR BODY" marine. Uses
//  marine_novr.iqm: the SAME 152-joint rig as marine_nohands.iqm (identical joint names +
//  order, verified) but WITH hand geometry skinned onto the wrist bones -- so the native arm
//  IK poses CONNECTED hands for free, no separate hand model / scale / offset. The separate
//  controller-tracked vhand HUD hand is retired (set vr_show_hands 0) to avoid double hands.
//  Rigged 152-joint
//  IQM with 0 baked anims -- the native two-bone arm IK (VR_UpdateArmIK, p_user.cpp)
//  drives the arm bones every tic by writing modelData->proceduralPose.
//    * `modelsareattachments` is REQUIRED: it is the upload gate at models.cpp:635 that
//      actually pushes the IK-posed bones to the GPU. +DECOUPLEDANIMATIONS alone FAILS on
//      a 0-animation IQM (frame1 stays -1, CalculateBones is skipped) -- do NOT use it here.
//    * Living frames (PLAY A-F: spawn/walk/attack/pain) map to the model. Death frames
//      (PLAY G+) are intentionally left unmapped so death still animates as the sprite.
//    * When VR IK is inactive (flatscreen / before first solve) useProceduralPose is false
//      and the mesh renders at its bind pose (models.cpp GetBasePose) -- still a valid marine.
//  A prior commit hard-excluded PlayerPawn from voxel replacement to protect this model.
// ============================================================================
Model DoomPlayer
{
   Path "models/marine"
   Model 0 "marine_novr.iqm"
   // [XR] marine_novr.iqm is a 7-surface DOOM Eternal Slayer rip -- NOT a shared atlas. Each surface is an
   // INDEPENDENT Source bodypart with its OWN UV unwrap on its OWN dedicated part texture. Proven: the IQM
   // surface triangle counts (3212/4274/262/6322/1324/2568/240) map 1:1 to the decompiled SMD material
   // groups (arms=2 skins, torso=head+torso, shoulders, pouches/shells=legs), and the SMD/QC $cdmaterials +
   // $texturegroup reference a DIFFERENT texture per part. Every surface's UVs span ~[0,1] because each part
   // is unwrapped across the full 0..1 of ITS OWN image -- overlapping ranges are NOT a shared atlas.
   // (A prior "shared atlas" theory pointed all 7 at marine_skin.png, which is byte-identical to the TORSO
   // skin marine_torso.png -- that painted the whole body with the torso texture = the smear. Reverted.)
   // SurfaceSkin <modelIndex> <surfaceIndex> <file> -- per-SURFACE skin (MD3_MAX_SURFACES=32). Surface order
   // after the legs-mesh removal: 0=arm_R 1=arm_L 2=head(neck stump) 3=torso 4=shoulders 5=legs(pouches)
   // 6=legs(shells). NOT "Skin <n>" (per-MODEL, overflows -> "Too many models in DoomPlayer").
   SurfaceSkin 0 0 "marine_arm_right.png"
   SurfaceSkin 0 1 "marine_arm_left.png"
   SurfaceSkin 0 2 "marine_head.png"
   SurfaceSkin 0 3 "marine_torso.png"
   SurfaceSkin 0 4 "marine_shoulders.png"
   SurfaceSkin 0 5 "marine_legs.png"
   SurfaceSkin 0 6 "marine_legs.png"
   Scale 1.0 1.0 1.0
   modelsareattachments

   FrameIndex PLAY A 0 0
   FrameIndex PLAY B 0 0
   FrameIndex PLAY C 0 0
   FrameIndex PLAY D 0 0
   FrameIndex PLAY E 0 0
   FrameIndex PLAY F 0 0
}

// VR Sword blade models -- static meshes (no armature), generated via Blender + the IQM
// exporter at Desktop\iqm-master\blender-4.1\iqm_export.py. Zero baked frames is expected
// and safe for a static mesh: models_iqm.cpp:210-212 explicitly falls back to a single
// implicit frame when num_frames==0 ("If a model doesn't have an animation loaded, it will
// crash. We don't want that!"), which with zero joints just renders the raw static mesh --
// this is NOT the same bug class as the rigged VR hand's zero-baked-frames issue (that
// model has real joints expecting real poses; these have none).
// One FrameIndex line each is enough: vr_sword.zs reuses sprite letter "A" across every
// weapon state (Ready/Deselect/Select/Fire/Spawn), matching ShieldSaw's SHLD/SHSW convention.
// [XR] ZOffset -5 REMOVED file-wide (was here on every weapon block): it was blind-copied from
// the Chainsaw after user feedback that "chainsaw was perfect", without accounting for
// models.cpp:452 dividing ZOffset by the weapon's OWN Scale before applying it -- so the same
// raw "-5" produced a different real-world shift per weapon (chainsaw Scale 1.5 vs pistol 1.0 vs
// rifle 1.35, etc), which is why weapons render mispositioned/angled at rest. The proven-good
// reference mod (D:\...\1.0b_Weapons_VanillaVRPlus_v1.2\Modeldef.Doom.Weapons.def) uses ZERO
// ZOffset on every weapon -- restored to match. A forward Offset in the Y slot (weapon models point FORWARD = +Y --
// verified this session: the IQM exporter does no axis swap and GZDoom's IQM loader does no
// swizzle, so the blades are authored blade-along-+Y with the grip centered on the origin so
// the hand holds the GRIP). Chainsaw uses Offset 0 14 0 / Scale -1.5, but its mesh origin sits
// at the handle and its mesh is small; these blades already have grip-at-origin and are authored
// at map-unit size, so a smaller forward Offset and Scale 1.0 are the right starting point.
// All of Offset/ZOffset/Scale are headset-tunable (also via vr_3dweaponOffsetX/Y/Z, vr_weaponScale).

// fires (the very first Select/raise) and for the dropped-in-world Spawn actor -- neither of which
// has a per-actor model override yet -- the lookup is FindModelFrame(VRSword, VRSW, A). Without

// off; JGRN->GRND is an exact 1:1 letter match (8 letters each), no guessing involved there.
Model HandGrenade
{
   Path "models/weapons/hud/grenade"
   Model 0 "nade.md3"
   Skin 0 "nade.png"
   Scale -1.0 1.0 1.0
   Offset 0.0 0.0 0.0
   NoInterpolation
   ZOffset -10

	// [XR] Verified against nade.md3's own bounding boxes: frame 0 is a mid-transition pose
	// (span shrinks 0->1->2, settling at frame 2/3), NOT a static resting pose -- same bug
	// class as the Pistol. B/C/D below are left at 1/2/3 -- those are an ACTIVE "pull pin"
	// windup during Fire, where a progression is intentional, not a static hold like A.
	// Ready / Deselect / Select (JHND A) -- was reference frame 0 (GRHO A); frame 0 is transitional
	FrameIndex	JHND	A	0	2

	// Fire: pull-pin windup (JHND B/C/D) -- reference frames 1-3 (GRHO B/C aim, GRTH A throw-start)
	FrameIndex	JHND	B	0	1
	FrameIndex	JHND	C	0	2
	FrameIndex	JHND	D	0	3

	// Throw: windup + release (JHND E/F) -- reference frames 6, 9 (GRTH D mid-swing, GRTH G release)
	FrameIndex	JHND	E	0	6
	FrameIndex	JHND	F	0	9

	// Throw: follow-through (JHND G/H/I) -- reference frames 11, 13, 15 (GRTH I/K/M settle)
	FrameIndex	JHND	G	0	11
	FrameIndex	JHND	H	0	13
	FrameIndex	JHND	I	0	15

	// Spawn/world-pickup sprite (JGRN, weapongrenade.zs) has ZERO graphic lumps anywhere -- this
	// left sprites[JGRN].numframes==0, which p_mobj.cpp's SpawnMapThing treats as "no frames" and
	// SILENTLY REPLACES the spawned actor with the generic Unknown placeholder. Since
	// vr_weapon_logic.zs gives a live 20% chance to replace map RocketAmmo/RocketBox with
	// HandGrenade, this was corrupting ~1 in 5 rocket-ammo pickups into a non-functional prop on
	// every map. Mapping to the ready pose (frame 0) fixes both the render AND the numframes==0
	// map-thing substitution bug in one line.
	FrameIndex	JGRN	A	0	0
}
