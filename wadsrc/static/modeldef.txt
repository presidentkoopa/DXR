// Reference frame layout (from the source asset's own Modeldef.Grenade.def) preserved in
// comments; FrameIndex letters remapped from the reference's GRHO/GRTH/GRND sprites to our
// actual HandGrenade weapon's real sprites (JHND in-hand, JGRN thrown -- see weapongrenade.zs).
// The JHND mapping below is a first-pass timing estimate (9 real states spread across the
// reference's 16-frame throw animation) -- sanity-check in VR and retune if the motion reads
// off; JGRN->GRND is an exact 1:1 letter match (8 letters each), no guessing involved there.
Model HandGrenade
{
   Path "models/weapons/hud/grenade"
   Model 0 "nade.md3"
   Skin 0 "nade.png"
   Scale -1.0 1.0 1.0
   Offset 0.0 0.0 0.0
   NoInterpolation
   ZOffset -10

	// Ready / Deselect / Select (JHND A) -- reference frame 0 (GRHO A, ready pose)
	FrameIndex	JHND	A	0	0

	// Fire: pull-pin windup (JHND B/C/D) -- reference frames 1-3 (GRHO B/C aim, GRTH A throw-start)
	FrameIndex	JHND	B	0	1
	FrameIndex	JHND	C	0	2
	FrameIndex	JHND	D	0	3

	// Throw: windup + release (JHND E/F) -- reference frames 6, 9 (GRTH D mid-swing, GRTH G release)
	FrameIndex	JHND	E	0	6
	FrameIndex	JHND	F	0	9

	// Throw: follow-through (JHND G/H/I) -- reference frames 11, 13, 15 (GRTH I/K/M settle)
	FrameIndex	JHND	G	0	11
	FrameIndex	JHND	H	0	13
	FrameIndex	JHND	I	0	15
}
